# Binary search algorithm from sorted list
def binary_search(arr, x):
    lo = 0
    hi = len(arr) - 1
    mid = 0

    while lo <= hi:
        mid = (hi + lo) // 2
        if arr[mid] < x:
            lo = mid + 1
        elif arr[mid] > x:
            hi = mid - 1
        else:
            return mid
    return -1
